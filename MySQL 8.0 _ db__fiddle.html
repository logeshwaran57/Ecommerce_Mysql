-- COMPLETE ECOMMERCE PROJECT EXECUTION
-- =====================================
-- Current Date: 2026-02-09 (for 30-day query reference)

-- 1. Create database and use it
CREATE DATABASE IF NOT EXISTS ecommerce;
USE ecommerce;

-- 2. Drop tables if exist (for clean re-run)
DROP TABLE IF EXISTS order_items;
DROP TABLE IF EXISTS orders;
DROP TABLE IF EXISTS products;
DROP TABLE IF EXISTS customers;

-- 3. Create tables
CREATE TABLE customers (
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100) NOT NULL,
email VARCHAR(100) UNIQUE NOT NULL,
address TEXT
);

CREATE TABLE products (
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(100) NOT NULL,
price DECIMAL(10,2) NOT NULL,
description TEXT
);

CREATE TABLE orders (
id INT AUTO_INCREMENT PRIMARY KEY,
customer_id INT,
order_date DATE NOT NULL,
total_amount DECIMAL(10,2) NOT NULL,
FOREIGN KEY (customer_id) REFERENCES customers(id)
);

-- REQUIRED QUERIES (Execute in order)

-- QUERY 1: Customers with orders in last 30 days
SELECT 'QUERY 1: Customers (last 30 days)' as Result;
SELECT DISTINCT c.name, c.email
FROM customers c JOIN orders o ON c.id = o.customer_id
WHERE o.order_date >= '2026-01-10'; -- 30 days before 2026-02-09

-- QUERY 2: Total amount per customer
SELECT 'QUERY 2: Total per customer' as Result;
SELECT c.name, SUM(o.total_amount) as total_spent
FROM customers c LEFT JOIN orders o ON c.id = o.customer_id
GROUP BY c.id, c.name;

-- QUERY 3: Update Product C price to 45.00
UPDATE products SET price = 45.00 WHERE name = 'Product C';
